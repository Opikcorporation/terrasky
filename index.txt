<script>
    document.addEventListener('DOMContentLoaded', () => {
        const form = document.getElementById('strategic-audit-form');
        const inputs = Array.from(form.querySelectorAll('.listenable'));
        const progressBar = document.getElementById('progress-bar');
        
        const filledState = new Map();

        const REACTIONS = [
            { emoji: '‚ú®', text: 'Super !' },
            { emoji: 'üëç', text: 'Bien not√©.' },
            { emoji: 'üéØ', text: 'Pr√©cis !' },
            { emoji: 'ü§î', text: 'Int√©ressant...' },
            { emoji: 'üëå', text: 'Parfait.' },
            { emoji: '‚úÖ', text: 'C\'est not√©.' }
        ];

        const updateProgressBar = () => {
            const requiredInputs = inputs.filter(el => el.required);
            const requiredFieldNames = new Set(requiredInputs.map(el => el.name || el.id));
            const filledCount = Array.from(filledState.keys()).filter(key => requiredFieldNames.has(key) && filledState.get(key)).length;
            const progress = requiredFieldNames.size > 0 ? (filledCount / requiredFieldNames.size) * 100 : 0;
            progressBar.style.width = `${progress}%`;
        };
        
        const getReactionContainer = (element) => {
            return element.closest('.form-group').querySelector('.feedback-reaction');
        };
        
        const showFeedback = (element) => {
            const reactionContainer = getReactionContainer(element);
            if (!reactionContainer || reactionContainer.classList.contains('visible')) return;
            
            const randomReaction = REACTIONS[Math.floor(Math.random() * REACTIONS.length)];
            reactionContainer.innerHTML = `<span>${randomReaction.emoji}</span> ${randomReaction.text}`;
            reactionContainer.classList.add('visible');

            setTimeout(() => {
                reactionContainer.classList.remove('visible');
            }, 2000);
        };

        const checkField = (element) => {
            let isFilled = false;
            let fieldName = element.name || element.id;

            if (element.type === 'radio') {
                const radioGroup = form.querySelectorAll(`input[name="${fieldName}"]`);
                if (Array.from(radioGroup).some(radio => radio.checked)) isFilled = true;
            } else if (element.type === 'checkbox') {
                 const checkboxGroup = form.querySelectorAll(`input[name="${fieldName}"]`);
                 if (Array.from(checkboxGroup).some(checkbox => checkbox.checked)) isFilled = true;
            } else if (element.value.trim() !== '') {
                isFilled = true;
            }

            if(isFilled && !filledState.get(fieldName)) {
                if(element.type !== 'radio' && element.type !== 'checkbox') {
                    showFeedback(element);
                }
            }
            
            filledState.set(fieldName, isFilled);
        };

        inputs.forEach(input => {
            const eventType = (input.type === 'radio' || input.type === 'checkbox') ? 'change' : 'blur';
            input.addEventListener(eventType, (e) => {
                checkField(e.target);
                updateProgressBar();
            });
        });

        const otherToggles = form.querySelectorAll('input[value="Autre"]');
        otherToggles.forEach(toggle => {
            const group = toggle.closest('.options-group');
            const otherInput = group.querySelector('.other-input');
            if (!otherInput) return;

            const groupName = toggle.name;

            group.addEventListener('change', (e) => {
                if (e.target.name !== groupName) return;
                const isOtherSelected = toggle.checked;

                if (isOtherSelected) {
                    otherInput.classList.add('visible');
                    otherInput.focus();
                } else {
                     otherInput.classList.remove('visible');
                     otherInput.value = '';
                }
            });
        });

        inputs.forEach(input => checkField(input));
        updateProgressBar();
    });
</script>
